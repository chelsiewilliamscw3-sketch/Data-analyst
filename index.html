SQL Snippets

WITH m AS (
  SELECT YEAR([Date]) AS [Year], DATEPART(QUARTER, [Date]) AS [Quarter],
         Revenue, COGS, OpEx
  FROM dbo.Finance
)
SELECT [Year], [Quarter],
       SUM(Revenue) AS Revenue,
       SUM(Revenue-COGS-OpEx) AS NetIncome
FROM m
GROUP BY [Year], [Quarter]
ORDER BY [Year], [Quarter];
